Total GDP = SUM(GDP[GDP])
Changed the type to currency and removed extra decimal numbers

Avg Internet Users = AVERAGE(InternetUsage[No. of Internet Users])

Selected Year = 
IF(
    HASONEVALUE('Year Table'[Year]),
    VALUES('Year Table'[Year]),
    "All Years"
)


YoY GDP Growth % = 
VAR CurrentYearGDP = [Total GDP]
VAR PreviousYr = MAX(GDP[Year]) - 1
VAR PreviousYearGDP = 
    CALCULATE(
        [Total GDP],
        GDP[Year] = PreviousYr
    )
RETURN
    IF(
        NOT(ISBLANK(PreviousYearGDP)),
        DIVIDE(CurrentYearGDP - PreviousYearGDP, PreviousYearGDP) * 100,
        BLANK()
    )


YoY Internet Growth % = 
VAR CurrentYear = [Avg Internet Users]
VAR PreviousYr = MAX(InternetUsage[Year]) - 1
VAR PreviousYearInternet = 
    CALCULATE(
        [Avg Internet Users],
        InternetUsage[Year] = PreviousYr
    )
RETURN
    IF(
        NOT(ISBLANK(PreviousYearInternet)),
        DIVIDE(CurrentYear - PreviousYearInternet, PreviousYearInternet) * 100,
        BLANK()
    )
